module Main where
import "base" Data.Traversable
import "transformers" Data.Functor.Reverse
import "transformers" Control.Monad.Trans.State
type Current = Int
type Max = Int
walls :: [Current]
walls = [2,5,1,2,3,4,7,7,6]
themax :: Current -> State Max Max
themax x = modify (max x) *> get
main = do
let lms = evalState (for walls themax) 0
let Reverse rms = evalState (for (Reverse walls) themax) 0
print . sum $ zipWith3 (\l x r -> min l r - x) lms walls rms

  • II этап. Онлайн-тест

    В день начала отбора отправим на почту ссылку на тест. В нём 130 вопросов на 6 тем. На выполнение — 3 дня. Результат оцениваем в баллах, которые учитываем при поступлении.

    Как проходит:
    • Чтение и написание кода на Java
    • Проверка домашнего задания
    • Разговор с преподавателем курса
    Для подготовки к разделу теста «программирование», а также к собеседованию, регистрируйся и готовься к тестированию.
